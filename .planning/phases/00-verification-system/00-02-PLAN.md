  <action>
    Create a new composable at `app/composables/useVerification.ts` with the following structure:

    ```typescript
    import type { VerificationResult, ValidateResult } from '~/server/services/verificationService'

    export interface VerificationOptions {
      obraId: string
      email?: string
      phone?: string
      method: 'email' | 'sms' | 'both'
    }

    export function useVerification() {
      // Reactive state
      const verifying = ref(false)
      const sending = ref(false)
      const canResend = ref(true)
      const resendCountdown = ref(0)
      const lastSentAt = ref<Date | null>(null)
      const error = ref<string | null>(null)
      const verified = ref(false)

      // Send verification code
      async function sendCode(options: VerificationOptions) {
        sending.value = true
        error.value = null

        try {
          const response = await $fetch<{ success: boolean; data: { message: string; method: string; expiresAt: string } }>('/api/verification/generate', {
            method: 'POST',
            body: options
          })

          lastSentAt.value = new Date()
          startResendCountdown()

          // Return response directly - API already returns correct structure: { success, data }
          return response
        } catch (err: any) {
          const message = err.data?.message || err.message || 'Error al enviar código de verificación'
          error.value = message
          return { success: false, error: message }
        } finally {
          sending.value = false
        }
      }

      // Validate verification code
      async function validateCode(code: string, obraId: string) {
        verifying.value = true
        error.value = null

        try {
          const response = await $fetch<{ success: boolean; data: { verified: boolean } }>('/api/verification/validate', {
            method: 'POST',
            body: { code, obraId }
          })

          if (response.success && response.data.verified) {
            verified.value = true
          }

          return response
        } catch (err: any) {
          const message = err.data?.message || err.message || 'Código de verificación inválido'
          error.value = message
          return { success: false, error: message }
        } finally {
          verifying.value = false
        }
      }

      // Start resend countdown (60 seconds)
      function startResendCountdown() {
        canResend.value = false
        resendCountdown.value = 60

        const interval = setInterval(() => {
          resendCountdown.value--
          if (resendCountdown.value <= 0) {
            clearInterval(interval)
            canResend.value = true
          }
        }, 1000)
      }

      // Reset verification state
      function reset() {
        verifying.value = false
        sending.value = false
        canResend.value = true
        resendCountdown.value = 0
        lastSentAt.value = null
        error.value = null
        verified.value = false
      }

      return {
        // State
        verifying,
        sending,
        canResend,
        resendCountdown,
        lastSentAt,
        error,
        verified,

        // Methods
        sendCode,
        validateCode,
        reset
      }
    }
    ```

    Key requirements:
    - Use $fetch for API calls (Nuxt 3 pattern)
    - All error messages in Spanish
    - 60-second resend countdown
    - TypeScript types for request/response
    - Follow existing composable patterns (see useS3.ts for reference)
    - Use ref() for reactive state
    - Handle error responses with proper type checking
