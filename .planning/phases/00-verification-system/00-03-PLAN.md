  <files>app/components/VerificationForm.vue</files>
  <action>
    Create a new component at `app/components/VerificationForm.vue` with the following structure:

    ```vue
    <template>
      <UCard>
        <template #header>
          <h3 class="text-lg font-medium">Verificación de Identidad</h3>
          <p class="text-sm text-gray-500 dark:text-gray-400">
            Verifica tu identidad mediante email o SMS
          </p>
        </template>

        <div class="space-y-4">
          <!-- Verification Method Selection -->
          <UFormField label="Método de verificación" name="method" required>
            <USelect
              v-model="selectedMethod"
              :items="methodOptions"
              placeholder="Selecciona cómo recibir tu código"
              :disabled="sending || verified"
            />
          </UFormField>

          <!-- Contact Info Display (read-only) -->
          <UAlert
            icon="i-heroicons-information-circle"
            color="info"
            variant="soft"
            :title="`Código enviado a ${selectedMethod === 'email' ? 'email' : 'SMS'}`"
            :description="selectedMethod === 'email' ? emailMask : phoneMask"
          />

          <!-- Verification Status -->
          <div v-if="verified" class="flex items-center gap-2 text-sm text-green-600 dark:text-green-400">
            <UIcon name="i-heroicons-check-circle" class="w-5 h-5" />
            <span>Verificación completada</span>
          </div>

          <!-- Code Input (show after sending) -->
          <div v-if="codeSent && !verified" class="space-y-4">
            <UFormField label="Código de verificación" name="code" required>
              <UInput
                v-model="code"
                placeholder="123456"
                maxlength="6"
                size="lg"
                :disabled="verifying"
                class="text-center text-2xl tracking-widest font-mono"
                @input="error = null"
              />
            </UFormField>

            <!-- Validate Button -->
            <UButton
              block
              color="primary"
              size="lg"
              :loading="verifying"
              :disabled="code.length !== 6"
              @click="handleValidate"
            >
              Verificar código
            </UButton>
          </div>

          <!-- Send/Resend Button -->
          <div v-if="!verified">
            <UButton
              v-if="!codeSent"
              block
              color="primary"
              size="lg"
              :loading="sending"
              :disabled="sending"
              @click="handleSendCode"
            >
              Enviar código de verificación
            </UButton>
            <UButton
              v-else
              block
              color="neutral"
              variant="outline"
              size="lg"
              :loading="sending"
              :disabled="sending || !canResend"
              @click="handleResend"
            >
              {{ canResend ? 'Reenviar código' : `Reenviar en ${resendCountdown}s` }}
            </UButton>
          </div>

          <!-- Error Display -->
          <UAlert
            v-if="error"
            icon="i-heroicons-exclamation-triangle"
            color="error"
            variant="soft"
            :title="error"
          />

          <!-- Help Text -->
          <UAlert
            icon="i-heroicons-information-circle"
            color="info"
            variant="soft"
            title="Información importante"
            description="El código de verificación expira en 15 minutos. Si no lo recibes, verifica tu carpeta de spam o solicita un nuevo código."
          />
        </div>
      </UCard>
    </template>

    <script setup lang="ts">
    import { useVerification } from '~/composables/useVerification'
    import type { VerificationOptions } from '~/composables/useVerification'

    interface Props {
      obraId: string
      email?: string
      phone?: string
    }

    const props = defineProps<Props>()

    const emit = defineEmits<{
      verified: [verified: boolean]
    }>()

    const toast = useToast()

    // Composable
    const { verifying, sending, canResend, resendCountdown, error, verified, sendCode, validateCode } = useVerification()

    // Local state
    const selectedMethod = ref<'email' | 'sms'>('email')
    const code = ref('')
    const codeSent = ref(false)

    // Method options
    const methodOptions = [
      { label: 'Email', value: 'email' },
      { label: 'SMS', value: 'sms' }
    ]

    // Mask contact info for display
    const emailMask = computed(() => {
      if (!props.email) return ''
      const [local, domain] = props.email.split('@')
      if (!domain) return props.email
      const maskedLocal = local.charAt(0) + '***' + local.charAt(local.length - 1)
      return `${maskedLocal}@${domain}`
    })

    const phoneMask = computed(() => {
      if (!props.phone) return ''
      return props.phone.replace(/(\d{3})\d{4}(\d{3})/, '$1****$2')
    })

    // Send verification code
    async function handleSendCode() {
      const options: VerificationOptions = {
        obraId: props.obraId,
        method: selectedMethod.value
      }

      if (selectedMethod.value === 'email' && props.email) {
        options.email = props.email
      } else if (selectedMethod.value === 'sms' && props.phone) {
        options.phone = props.phone
      } else {
        toast.add({
          title: 'Información requerida',
          description: `Debes proporcionar ${selectedMethod.value === 'email' ? 'un email' : 'un teléfono'}`,
          color: 'error',
          icon: 'i-heroicons-exclamation-triangle'
        })
        return
      }

      const result = await sendCode(options)

      if (result.success) {
        codeSent.value = true
        toast.add({
          title: 'Código enviado',
          description: `Revisa tu ${selectedMethod.value === 'email' ? 'email' : 'SMS'}`,
          color: 'success',
          icon: 'i-heroicons-check-circle'
        })
      }
    }

    // Validate verification code
    async function handleValidate() {
      if (code.value.length !== 6) {
        toast.add({
          title: 'Código inválido',
          description: 'El código debe tener 6 dígitos',
          color: 'error',
          icon: 'i-heroicons-exclamation-triangle'
        })
        return
      }

      const result = await validateCode(code.value, props.obraId)

      if (result.success && result.data?.verified) {
        toast.add({
          title: 'Verificación exitosa',
          description: 'Tu identidad ha sido verificada',
          color: 'success',
          icon: 'i-heroicons-check-circle'
        })
        emit('verified', true)
      }
    }

    // Resend code
    async function handleResend() {
      await handleSendCode()
    }

    // Watch for verified status changes
    watch(verified, (isVerified) => {
      if (isVerified) {
        emit('verified', true)
      }
    })

    // Expose verified status to parent
    defineExpose({
      verified
    })
    </script>
    ```

    Key requirements:
    - Use Nuxt UI v4 components (UCard, UFormField, UInput, USelect, UButton, UAlert, UIcon)
    - Email and SMS method selection
    - Mask contact info for privacy
    - Resend countdown with disabled button
    - Spanish text throughout
    - Emit 'verified' event when successfully verified
    - Handle error states with toast notifications
    - Support props for obraId, email, phone
  </action>
