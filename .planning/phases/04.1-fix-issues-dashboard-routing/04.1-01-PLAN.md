```vue
<template>
  <div class="container mx-auto px-4 py-8">
    <!-- Header -->
    <div class="mb-8">
      <div class="flex items-center justify-between mb-4">
        <div>
          <h1 class="text-3xl font-bold text-gray-900 dark:text-white">
            Incidencias de la Obra
          </h1>
          <p class="text-gray-600 dark:text-gray-400" v-if="plan">
            {{ plan.nom_obra || 'Sin nombre' }}
          </p>
        </div>
        <UButton
          icon="i-heroicons-arrow-left"
          variant="outline"
          @click="navigateTo(`/protected/planes/${planId}`)"
        >
          Volver al Dashboard
        </UButton>
      </div>
    </div>

    <!-- Loading State -->
    <div v-if="loading" class="flex justify-center py-12">
      <UIcon name="i-heroicons-arrow-path" class="w-8 h-8 animate-spin" />
    </div>

    <!-- Error State -->
    <div v-else-if="error" class="text-center py-12">
      <UIcon name="i-heroicons-exclamation-triangle" class="w-12 h-12 text-red-500 mx-auto mb-4" />
      <h3 class="text-lg font-medium text-gray-900 dark:text-white mb-2">
        Error cargando el plan
      </h3>
      <p class="text-gray-600 dark:text-gray-400 mb-4">
        {{ error }}
      </p>
      <UButton @click="loadPlan" variant="outline">
        Reintentar
      </UButton>
    </div>

    <!-- Issues Dashboard -->
    <ObraDashboard
      v-else-if="plan"
      :obra-id="plan._id"
      :obra-name="plan.nom_obra || 'Plan sin nombre'"
    />
  </div>
</template>

<script setup lang="ts">
import ObraDashboard from '~/components/ObraDashboard.vue'
import type { Plan } from '~/stores/planes'

const route = useRoute()
const planId = route.params.id as string

const loading = ref(true)
const error = ref<string | null>(null)
const plan = ref<Plan | null>(null)

async function loadPlan() {
  loading.value = true
  error.value = null

  try {
    const response = await $fetch<any>(`/api/planes/${planId}`)
    plan.value = response.data
  } catch (err: any) {
    error.value = err.data?.message || err.message || 'Error al cargar el plan'
  } finally {
    loading.value = false
  }
}

onMounted(() => {
  loadPlan()
})
</script>
```
