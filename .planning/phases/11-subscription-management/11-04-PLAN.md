   ```typescript

     // Implementation
   }
   ```

3. **State management**:
   - subscription: Ref<SubscriptionData | null> = ref(null)
   - loading: Ref<boolean> = ref(false)
   - error: Ref<string | null> = ref(null)

4. **Fetch subscription function**:
   ```typescript
   async function fetchSubscription() {
     loading.value = true
     error.value = null
     try {
       const response = await $fetch<{success: boolean, data: SubscriptionData}>(
         `/api/planes/${planId.value}/subscription`
       )
       if (response.success) {
         subscription.value = response.data
       }
     } catch (err: any) {
       // 404 is expected if no subscription exists
       if (err.statusCode === 404) {
         subscription.value = null
       } else {
         error.value = err.message || 'Failed to load subscription'
       }
     } finally {
       loading.value = false
     }
   }
   ```

5. **Pause subscription method**:
   ```typescript
   async function pauseSubscription(options: {indefinite?: boolean, resumeDate?: string}) {
     loading.value = true
     try {
       await $fetch(`/api/planes/${planId.value}/subscription/pause`, {
         method: 'POST',
         body: options
       })
       await fetchSubscription() // Refresh subscription data
     } catch (err: any) {
       error.value = err.message || 'Failed to pause subscription'
       throw err
     } finally {
       loading.value = false
     }
   }
   ```

6. **Resume subscription method**:
   ```typescript
   async function resumeSubscription() {
     loading.value = true
     try {
       await $fetch(`/api/planes/${planId.value}/subscription/resume`, {
         method: 'POST'
       })
       await fetchSubscription()
     } catch (err: any) {
       error.value = err.message || 'Failed to resume subscription'
       throw err
     } finally {
       loading.value = false
     }
   }
   ```

7. **Cancel subscription method**:
   ```typescript
   async function cancelSubscription() {
     loading.value = true
     try {
       await $fetch(`/api/planes/${planId.value}/subscription/cancel`, {
         method: 'POST'
       })
       await fetchSubscription()
     } catch (err: any) {
       error.value = err.message || 'Failed to cancel subscription'
       throw err
     } finally {
       loading.value = false
     }
   }
   ```

8. **Return reactive state and methods**:
   ```typescript
   return {
     subscription,
     loading,
     error,
     fetchSubscription,
     pauseSubscription,
     resumeSubscription,
     cancelSubscription
   }
   ```
