Add `openPortal` method to the `useSubscription` composable in `app/composables/useSubscription.ts`:

1. **Add openPortal function** after the `cancelSubscription` function:
   ```typescript
   /**
   * Open Stripe Customer Portal
   *
   * @remarks
   * Calls portal API endpoint and redirects user to Stripe Customer Portal.
   * Portal allows users to manage payment methods and subscriptions.
   * Throws error if portal session creation fails.
   */
   async function openPortal(): Promise<void> {
     loading.value = true
     error.value = null

     try {
       const response = await $fetch<{
         success: boolean
         data: { portalUrl: string; sessionId: string }
       }>(`/api/planes/${planId.value}/subscription/portal`, {
         method: 'POST'
       })

       if (response.success && response.data.portalUrl) {
         // Redirect to Stripe Customer Portal
         await navigateTo(response.data.portalUrl, { external: true })
       }
     } catch (err: any) {
       console.error('Failed to open portal:', err)

       // Handle error messages
       if (err.statusMessage) {
         error.value = err.statusMessage
       } else if (err.data?.message) {
         error.value = err.data.message
       } else {
         error.value = 'Failed to open payment portal'
       }

       throw err
     } finally {
       loading.value = false
     }
   }
   ```

2. **Update return statement** to include `openPortal`:
   ```typescript
   return {
     subscription,
     loading,
     error,
     fetchSubscription,
     pauseSubscription,
     resumeSubscription,
     cancelSubscription,
     openPortal  // ADD THIS
   }
   ```

**IMPORTANT:** This method calls the existing portal.post.ts endpoint from Plan 11-03 and redirects to the Stripe Customer Portal URL. The portal return handling is already implemented in dashboard.vue.
