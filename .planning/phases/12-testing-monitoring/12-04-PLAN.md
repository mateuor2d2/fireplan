    ```typescript
    /**
     * Checks if user has access to subscription (including grace period)
     *
     * Stripe Smart Retries automatically retry failed payments over ~3 weeks.
     * During this retry period, subscription status is 'past_due'.
     * Users should maintain access during grace period.
     *
     * @param subscription - Subscription object to check
     * @returns true if user has access (active or past_due), false if expired
     */
    export function hasSubscriptionAccess(subscription: ISubscription): boolean {
      // Active: Paid and current
      // Past due: Failed payment but retrying (grace period - Smart Retries)
      // Both should have access
      return subscription.status === 'active' || subscription.status === 'past_due'
    }

    /**
     * Checks if subscription is truly expired (access should be revoked)
     *
     * Only revoke access when subscription is canceled or incomplete_expired.
     * The 'past_due' status is a temporary grace period during retries.
     *
     * @param subscription - Subscription object to check
     * @returns true if access should be revoked, false otherwise
     */
    export function isSubscriptionExpired(subscription: ISubscription): boolean {
      return subscription.status === 'canceled' || subscription.status === 'incomplete_expired'
    }
    ```

    Import ISubscription type from '~/types/subscription'

    These functions document Stripe Smart Retry behavior:
    - active = fully paid, full access
    - past_due = temporary failure, retrying in progress (grace period)
    - canceled = user canceled or payment permanently failed
    - incomplete_expired = trial period ended without payment

    Key insight: Treat `past_due` as having access (don't revoke early).
