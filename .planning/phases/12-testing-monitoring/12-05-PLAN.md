       ```typescript
       async function handleInvoicePaymentFailure(invoice: any) {
         try {
           console.log('handleInvoicePaymentFailure called with invoice:', invoice.id)

           // Find invoice record in database
           const dbInvoice = await Invoice.findOne({
             stripeInvoiceId: invoice.id
           })

           if (!dbInvoice) {
             console.error('Invoice record not found for failed invoice:', invoice.id)
             return
           }

           // Get user information
           const user = await User.findById(dbInvoice.userId)
           if (!user) {
             console.error('User not found for invoice:', dbInvoice.userId)
             return
           }

           // Get plan information
           const plan = await Planes.findById(dbInvoice.planId)
           if (!plan) {
             console.error('Plan not found for invoice:', dbInvoice.planId)
             return
           }

           // Get invoice URL for payment update (Stripe-hosted invoice URL)
           const invoiceUrl = invoice.hosted_invoice_url || `https://dashboard.stripe.com/invoices/${invoice.id}`

           // Get retry count from Stripe invoice
           const retryCount = invoice.retries_remaining || 0

           // Send email notification
           await sendPaymentFailureEmail(
             user.email,
             plan.nom_obra,
             invoiceUrl,
             retryCount
           )

           console.log('Payment failure email sent for invoice:', invoice.id)
         } catch (error) {
           console.error('Error handling invoice payment failure:', error)
         }
       }
       ```
