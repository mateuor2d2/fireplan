# GSD Plan: Lazy Load PDFMake + Externalize Heavy Server Deps

## Objective
Reduce server bundle size by lazy-loading pdfmake and externalizing heavy dependencies

## Problem
- `pdfmake` (5MB+) bundled in server output
- Heavy server dependencies included in bundle unnecessarily
- PDF generation happens on-demand, no need to load at startup

## Current State
```bash
du -sh .output/server  # Currently 71MB
```

## Implementation

### Task 1: Create lazy-loaded PDF generator utility

Create `server/utils/pdf-generator-lazy.ts`:
```typescript
import type { TDocumentDefinitions } from 'pdfmake/interfaces'

let pdfMake: typeof import('pdfmake') | null = null
let vfsFonts: any = null

async function loadPdfMake() {
  if (!pdfMake) {
    const [pdfMakeModule, fontsModule] = await Promise.all([
      import('pdfmake'),
      import('./vfs_fonts.js')
    ])
    pdfMake = pdfMakeModule
    vfsFonts = fontsModule
  }
  return { pdfMake, vfsFonts }
}

export async function generatePdfLazy(docDefinition: TDocumentDefinitions): Promise<Buffer> {
  const { pdfMake, vfsFonts } = await loadPdfMake()
  
  return new Promise((resolve, reject) => {
    const printer = new pdfMake.Printer({
      ...vfsFonts.pdfMake
    })
    
    const pdfDoc = printer.createPdfKitDocument(docDefinition)
    const chunks: Buffer[] = []
    
    pdfDoc.on('data', chunk => chunks.push(chunk))
    pdfDoc.on('end', () => resolve(Buffer.concat(chunks)))
    pdfDoc.on('error', reject)
    
    pdfDoc.end()
  })
}
```
