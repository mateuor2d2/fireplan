# GSD Plan: Replace Sharp with Optional Dependency

## Objective
Make Sharp optional to reduce bundle size and handle environments where it fails to install

## Problem
- `sharp` is 17MB and requires native compilation
- Often fails on different platforms/deployments
- Not critical for basic image handling

## Implementation

### Task 1: Create image compression utility with fallback

Create `server/utils/imageCompression.ts`:
```typescript
let sharpModule: typeof import('sharp') | null = null

async function loadSharp() {
  if (sharpModule === null) {
    try {
      sharpModule = await import('sharp')
    } catch (e) {
      console.warn('Sharp not available, images will not be optimized')
      sharpModule = undefined
    }
  }
  return sharpModule
}

export async function compressImage(
  buffer: Buffer,
  options: {
    maxWidth?: number
    maxHeight?: number
    quality?: number
  } = {}
): Promise<Buffer> {
  const { maxWidth = 1200, maxHeight = 1200, quality = 80 } = options
  const sharp = await loadSharp()
  
  if (!sharp) {
    return buffer
  }
  
  try {
    return await sharp(buffer)
      .resize(maxWidth, maxHeight, {
        fit: 'inside',
        withoutEnlargement: true
      })
      .jpeg({ quality, mozjpeg: true })
      .toBuffer()
  } catch (error) {
    console.error('Image compression failed:', error)
    return buffer
  }
}

export async function getImageMetadata(buffer: Buffer) {
  const sharp = await loadSharp()
  
  if (!sharp) {
    return null
  }
  
  try {
    return await sharp(buffer).metadata()
  } catch {
    return null
  }
}

export function isSharpAvailable(): boolean {
  return sharpModule !== undefined
}
```

Update each file to use `imageCompression.ts` utilities.
